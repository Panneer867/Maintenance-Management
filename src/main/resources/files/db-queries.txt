For Checking Roles

SELECT DISTINCT U.USER_ID, U.ROLE_ID, RM.NAME AS ROLE_NAME, P.PRIVILEGE_ID, PP.NAME AS PRIVILEGE_NAME FROM USERS_ROLES U FULL OUTER JOIN ROLE_PRIVILEGES P ON U.ROLE_ID = P.ROLE_ID FULL OUTER JOIN ROLE_MASTER RM ON U.ROLE_ID = RM.ID FULL OUTER JOIN PRIVILEGE PP ON P.PRIVILEGE_ID = PP.ID ORDER BY U.USER_ID ASC
